68     @Inject
69     public JoyentCloudComputeServiceAdapter(JoyentCloudApi cloudApiApi, @Zone Supplier<Set<String>> datacenterIds) {
70        this.cloudApiApi = checkNotNull(cloudApiApi, "cloudApiApi");

74     @Override
75     public NodeAndInitialCredentials<MachineInDatacenter> createNodeWithGroupEncodedIntoName(String group, String name,
76           Template template) {
77  
78        LoginCredentials.Builder credentialsBuilder = LoginCredentials.builder();
79  
80        CreateMachineOptions options = new CreateMachineOptions();
81        options.name(name);
82        options.packageName(template.getHardware().getProviderId());
83        options.metadata(metadataAndTagsAsCommaDelimitedValue(template.getOptions()));
84  
85        String datacenterId = template.getLocation().getId();
86        String datasetURN = template.getImage().getProviderId();
87  
88        logger.debug(">> creating new machine datacenter(%s) datasetURN(%s) options(%s)", datacenterId, datasetURN,
89              options);
90        Machine machine = cloudApiApi.getMachineApiForDatacenter(datacenterId).createWithDataset(datasetURN, options);
91  
92        logger.trace("<< machine(%s)", machine.getId());
93  
94        MachineInDatacenter machineInDatacenter = new MachineInDatacenter(machine, datacenterId);
95  
96        //TODO machineInDatacenter.metadata for password
97        if (template.getOptions().getLoginPrivateKey() != null){
98           credentialsBuilder.privateKey(template.getOptions().getLoginPrivateKey());
99        }
100       return new NodeAndInitialCredentials<MachineInDatacenter>(machineInDatacenter, machineInDatacenter.slashEncode(),
101             credentialsBuilder.build());
102    }

104    @Override
105    public Iterable<PackageInDatacenter> listHardwareProfiles() {
106       Builder<PackageInDatacenter> builder = ImmutableSet.builder();
107       for (final String datacenterId : datacenterIds.get()) {
108          builder.addAll(transform(cloudApiApi.getPackageApiForDatacenter(datacenterId).list(),
109                new Function<org.jclouds.joyent.cloudapi.v6_5.domain.Package, PackageInDatacenter>() {
110 
111                   @Override
112                   public PackageInDatacenter apply(org.jclouds.joyent.cloudapi.v6_5.domain.Package arg0) {
113                      return new PackageInDatacenter(arg0, datacenterId);
114                   }
115 
116                }));
117       }
118       return builder.build();
119    }
