47  public class CloudWatchLiveTest extends BaseContextLiveTest<RestContext<CloudWatchApi, CloudWatchAsyncApi>> {
48  
49     public CloudWatchLiveTest() {
50        provider = "cloudwatch";
51     }
52  
53     private CloudWatchApi api;
54  
55     @Override
56     @BeforeClass(groups = { "integration", "live" })
57     public void setupContext() {
58        super.setupContext();
59        api = context.getApi();
60     }
61  
62     @Override
63     protected TypeToken<RestContext<CloudWatchApi, CloudWatchAsyncApi>> contextType() {
64        return CloudWatchApiMetadata.CONTEXT_TOKEN;
65     }
66  
67     @Test
68     protected void testCloudWatchListMetrics() {
69        // Just make sure there is at least one metric returned (Much better if the account you use has more than 500)
70        checkArgument(CloudWatch.listMetrics(api, null, new ListMetricsOptions()).iterator().hasNext());
71     }
72  
73     @Test
74     protected void testCloudWatchPutMetrics() {
75        String metricName = "TestMetricName" + System.currentTimeMillis();
76        String namespace = "JCLOUDS/Test";
77        Date metricTimestamp = new Date();
78        Set<MetricDatum> metrics = Sets.newLinkedHashSet();
79  
80        for (int i = 0; i < 11; i++) {
81           metrics.add(MetricDatum.builder()
82                                  .metricName(metricName + "_" + i)
83                                  .dimension(new Dimension("BaseMetricName", metricName))
84                                  .unit(Unit.COUNT)
85                                  .timestamp(metricTimestamp)
86                                  .value((double) i)
87                                  .build());
88        }
89  
90        CloudWatch.putMetricData(api, null, metrics, namespace);
91  
92        ListMetricsOptions lmo = ListMetricsOptions.Builder.namespace(namespace)
93                                                   .dimension(new Dimension("BaseMetricName", metricName));
94        boolean success = new RetryablePredicate<ListMetricsOptions>(new Predicate<ListMetricsOptions>() {
95           @Override
96           public boolean apply(ListMetricsOptions options) {
97              return Iterables.size(CloudWatch.listMetrics(api, null, options)) == 11;
98           }
99        }, 20, 1, TimeUnit.MINUTES).apply(lmo);
100 
101       if (!success) {
102          Assert.fail("Unable to gather the created CloudWatch data within the time (20m) allotted.");
103       }
104    }
