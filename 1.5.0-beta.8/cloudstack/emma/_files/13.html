243   @Override
244   public void destroyNode(String id) {
245      String virtualMachineId = id;
246      // There was a bug in 2.2.12 release happening when static nat IP address
247      // was being released, and corresponding firewall rules were left behind.
248      // So next time the same IP is allocated, it might be not be static nat
249      // enabled, but there are still rules associated with it. And when you try
250      // to release this IP, the release will fail.
251      //
252      // The bug was fixed in 2.2.13 release only, and the current system wasn't
253      // updated yet.
254      //
255      // To avoid the issue, every time you release a static nat ip address, do
256      // the following:
257 
258      // 1) Delete IP forwarding rules associated with IP.
259      Set<String> ipAddresses = deleteIPForwardingRulesForVMAndReturnDistinctIPs(virtualMachineId);
260 
261      // 2) Delete firewall rules associated with IP.
262      ipAddresses.addAll(deleteFirewallRulesForVMAndReturnDistinctIPs(virtualMachineId));
263 
264      // 3) Disable static nat rule for the IP.
265      disableStaticNATOnIPAddresses(ipAddresses);
266 
267      // 4) Only after 1 and 2 release the IP address.
268      disassociateIPAddresses(ipAddresses);
269 
270      destroyVirtualMachine(virtualMachineId);
271 
272      vmToRules.invalidate(virtualMachineId);
273   }
