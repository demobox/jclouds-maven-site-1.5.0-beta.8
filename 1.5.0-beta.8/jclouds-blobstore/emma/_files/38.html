454    * {@inheritDoc}
455    */
456   @Override
457   public ListenableFuture<Blob> getBlob(final String containerName, final String key, GetOptions options) {
458      logger.debug("Retrieving blob with key %s from container %s", key, containerName);
459      // If the container doesn't exist, an exception is thrown
460      if (!storageStrategy.containerExists(containerName)) {
461         logger.debug("Container %s does not exist", containerName);
462         return immediateFailedFuture(cnfe(containerName));
463      }
464      // If the blob doesn't exist, a null object is returned
465      if (!storageStrategy.blobExists(containerName, key)) {
466         logger.debug("Item %s does not exist in container %s", key, containerName);
467         return immediateFuture(null);
468      }
469 
470      Blob blob = loadBlob(containerName, key);
471 
472      if (options != null) {
473         if (options.getIfMatch() != null) {
474            if (!blob.getMetadata().getETag().equals(options.getIfMatch()))
475               return immediateFailedFuture(returnResponseException(412));
476         }
477         if (options.getIfNoneMatch() != null) {
478            if (blob.getMetadata().getETag().equals(options.getIfNoneMatch()))
479               return immediateFailedFuture(returnResponseException(304));
480         }
481         if (options.getIfModifiedSince() != null) {
482            Date modifiedSince = options.getIfModifiedSince();
483            if (blob.getMetadata().getLastModified().before(modifiedSince)) {
484               HttpResponse response = HttpResponse.builder().statusCode(304).build();
485               return immediateFailedFuture(new HttpResponseException(String.format("%1$s is before %2$s", blob
486                     .getMetadata().getLastModified(), modifiedSince), null, response));
487            }
488 
489         }
490         if (options.getIfUnmodifiedSince() != null) {
491            Date unmodifiedSince = options.getIfUnmodifiedSince();
492            if (blob.getMetadata().getLastModified().after(unmodifiedSince)) {
493               HttpResponse response = HttpResponse.builder().statusCode(412).build();
494               return immediateFailedFuture(new HttpResponseException(String.format("%1$s is after %2$s", blob
495                     .getMetadata().getLastModified(), unmodifiedSince), null, response));
496            }
497         }
498         blob = copyBlob(blob);
499 
500         if (options.getRanges() != null && options.getRanges().size() > 0) {
501            byte[] data;
502            try {
503               data = toByteArray(blob.getPayload());
504            } catch (IOException e) {
505               return immediateFailedFuture(new RuntimeException(e));
506            }
507            ByteArrayOutputStream out = new ByteArrayOutputStream();
508            for (String s : options.getRanges()) {
509               // HTTP uses a closed interval while Java array indexing uses a
510               // half-open interval.
511               int offset = 0;
512               int last = data.length - 1;
513               if (s.startsWith("-")) {
514                  offset = last - Integer.parseInt(s.substring(1)) + 1;
515               } else if (s.endsWith("-")) {
516                  offset = Integer.parseInt(s.substring(0, s.length() - 1));
517               } else if (s.contains("-")) {
518                  String[] firstLast = s.split("\\-");
519                  offset = Integer.parseInt(firstLast[0]);
520                  last = Integer.parseInt(firstLast[1]);
521               } else {
522                  return immediateFailedFuture(new IllegalArgumentException("illegal range: " + s));
523               }
524 
525               if (offset > last) {
526                  return immediateFailedFuture(new IllegalArgumentException("illegal range: " + s));
527               }
528               if (last + 1 > data.length) {
529                  last = data.length - 1;
530               }
531               out.write(data, offset, last - offset + 1);
532            }
533            ContentMetadata cmd = blob.getPayload().getContentMetadata();
534            byte[] byteArray = out.toByteArray();
535            blob.setPayload(byteArray);
536            HttpUtils.copy(cmd, blob.getPayload().getContentMetadata());
537            blob.getPayload().getContentMetadata().setContentLength(Long.valueOf(byteArray.length));
538         }
539      }
540      checkNotNull(blob.getPayload(), "payload " + blob);
541      return immediateFuture(blob);
542   }
