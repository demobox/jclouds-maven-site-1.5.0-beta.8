19  package org.jclouds.scriptbuilder.functionloader.osgi;
20  
21  import java.io.IOException;
22  import java.net.URL;
23  import java.util.Enumeration;
24  import java.util.Properties;
25  
26  import org.jclouds.scriptbuilder.domain.OsFamily;
27  import org.jclouds.scriptbuilder.domain.ShellToken;
28  import org.jclouds.scriptbuilder.functionloader.FunctionLoader;
29  import org.jclouds.scriptbuilder.functionloader.FunctionNotFoundException;
30  import org.osgi.framework.Bundle;
31  import org.osgi.framework.BundleContext;
32  import org.osgi.framework.ServiceRegistration;
33  
34  import com.google.common.base.Charsets;
35  import com.google.common.io.CharStreams;
36  import com.google.common.io.Resources;
37  
38  /**
39   * A {@link FunctionLoader} which searches for functions in the {@link Bundle} resources.
40   */
41  public class BundleFunctionLoader implements FunctionLoader {
42  
43     private final BundleContext bundleContext;
44     private ServiceRegistration registration;
45  
46     /**
47      * Constructor
48      * 
49      * @param bundleContext
50      */
51     public BundleFunctionLoader(BundleContext bundleContext) {
52        this.bundleContext = bundleContext;
53     }
54  
55     /**
56      * Starts the loader. Looks up for {@link Bundle} resources and registers itself in the service
57      * registry. It adds a property to the service which advertise all the functions found in the
58      * local resources.
59      */
60     public void start() {
61        Bundle bundle = bundleContext.getBundle();
62        Enumeration<?> entries = bundle.findEntries("/functions/", "*.*", false);
63        StringBuilder sb = new StringBuilder();
64        while (entries.hasMoreElements()) {
65           URL url = (URL) entries.nextElement();
66           String function = url.getFile();
67           sb.append(function);
68           if (entries.hasMoreElements()) {
69              sb.append(" ");
70           }
71        }
72        String functions = sb.toString();
73        registerFunction(functions);
74     }
75  
76     /**
77      * Unregisters itself from the service registry.
78      */
79     public void stop() {
80        registration.unregister();
81     }
82  
83     /**
84      * Loads the function from the {@link Bundle} resources.
85      * 
86      * @param function
87      *           The function name to load.
88      * @param family
89      *           This operating system family of the function.
90      * @return
91      * @throws FunctionNotFoundException
92      */
93     @Override
94     public String loadFunction(String function, OsFamily family) throws FunctionNotFoundException {
95        try {
96           return CharStreams.toString(Resources.newReaderSupplier(bundleContext.getBundle().getResource(
97                    String.format("/functions/%s.%s", function, ShellToken.SH.to(family))), Charsets.UTF_8));
98        } catch (IOException e) {
99           throw new FunctionNotFoundException(function, family, e);
100       }
101    }
102 
103    private void registerFunction(String functions) {
104       Properties properties = new Properties();
105       properties.put("function", functions);
106       registration = bundleContext.registerService(FunctionLoader.class.getName(), this, properties);
107    }
