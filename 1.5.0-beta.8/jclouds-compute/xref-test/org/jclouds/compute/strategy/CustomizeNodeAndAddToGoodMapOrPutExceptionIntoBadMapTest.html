199    public void testRecoversWhenTemporarilyNodeNotFound() {
200       String nodeId = "myid";
201 
202       InitializeRunScriptOnNodeOrPlaceInBadMap.Factory initScriptRunnerFactory = createMock(InitializeRunScriptOnNodeOrPlaceInBadMap.Factory.class);
203       OpenSocketFinder openSocketFinder = createMock(OpenSocketFinder.class);
204       Timeouts timeouts = new Timeouts();
205       Function<TemplateOptions, Statement> templateOptionsToStatement = new TemplateOptionsToStatement();
206       Set<NodeMetadata> goodNodes = Sets.newLinkedHashSet();
207       Map<NodeMetadata, Exception> badNodes = Maps.newLinkedHashMap();
208       Multimap<NodeMetadata, CustomizationResponse> customizationResponses = LinkedHashMultimap.create();
209       TemplateOptions options = new TemplateOptions();
210       
211       final NodeMetadata pendingNode = new NodeMetadataBuilder().ids(nodeId).status(Status.PENDING).build();
212       final NodeMetadata runningNode = new NodeMetadataBuilder().ids(nodeId).status(Status.RUNNING).build();
213       GetNodeMetadataStrategy nodeClient = createMock(GetNodeMetadataStrategy.class);
214       AtomicNodeRunning nodeRunning = new AtomicNodeRunning(nodeClient);
215       Predicate<AtomicReference<NodeMetadata>> retryableNodeRunning = new ComputeServiceTimeoutsModule() {
216                public Predicate<AtomicReference<NodeMetadata>> nodeRunning(AtomicNodeRunning statusRunning, Timeouts timeouts) {
217                   return super.nodeRunning(statusRunning, timeouts);
218                }
219             }.nodeRunning(nodeRunning, timeouts);
220       AtomicReference<NodeMetadata> atomicNode = new AtomicReference<NodeMetadata>(pendingNode);
221       
222       // Simulate transient error: first call returns null; subsequent calls return the running node
223       EasyMock.expect(nodeClient.getNode(nodeId)).andAnswer(new IAnswer<NodeMetadata>() {
224                private int count = 0;
225                @Override
226                public NodeMetadata answer() throws Throwable {
227                   count++;
228                   if (count <= 1) {
229                      return null;
230                   } else {
231                      return runningNode;
232                   }
233                }
234             }).anyTimes();
235 
236       // replay mocks
237       replay(initScriptRunnerFactory, openSocketFinder, nodeClient);
238       
239       // run
240       new CustomizeNodeAndAddToGoodMapOrPutExceptionIntoBadMap(retryableNodeRunning, openSocketFinder, timeouts,
241                templateOptionsToStatement, initScriptRunnerFactory, options, atomicNode, goodNodes, badNodes,
242                customizationResponses).apply(atomicNode);
243 
244       if (badNodes.size() > 0) Iterables.get(badNodes.values(), 0).printStackTrace();
245       assertEquals(badNodes.size(), 0);
246       assertEquals(goodNodes, ImmutableSet.of(runningNode));
247       assertEquals(customizationResponses.size(), 0);
248 
249       // verify mocks
250       verify(initScriptRunnerFactory, openSocketFinder, nodeClient);
251    }
