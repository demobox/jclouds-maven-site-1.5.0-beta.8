45     public static enum State {
46        STARTING, RUNNING, STOPPING, STOPPED, DESTROYED, EXPUNGING, MIGRATING, ERROR, UNKNOWN, SHUTDOWNED, UNRECOGNIZED;
47  
48        @Override
49        public String toString() {
50           return CaseFormat.UPPER_UNDERSCORE.to(CaseFormat.UPPER_CAMEL, name());
51        }
52  
53        public static State fromValue(String state) {
54           try {
55              return valueOf(CaseFormat.UPPER_CAMEL.to(CaseFormat.UPPER_UNDERSCORE, checkNotNull(state, "state")));
56           } catch (IllegalArgumentException e) {
57              return UNRECOGNIZED;
58           }
59        }
60  
61     }
