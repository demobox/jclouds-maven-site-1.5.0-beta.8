54  @Singleton
55  public class ELBLoadBalanceNodesStrategy implements LoadBalanceNodesStrategy {
56     @Resource
57     @Named(LoadBalancerConstants.LOADBALANCER_LOGGER)
58     protected Logger logger = Logger.NULL;
59     protected final ELBApi api;
60     protected final Function<LoadBalancerInRegion, LoadBalancerMetadata> converter;
61  
62     @Inject
63     protected ELBLoadBalanceNodesStrategy(ELBApi api,
64              Function<LoadBalancerInRegion, LoadBalancerMetadata> converter) {
65        this.api = checkNotNull(api, "api");
66        this.converter = checkNotNull(converter, "converter");
67     }
68  
69     @Override
70     public LoadBalancerMetadata createLoadBalancerInLocation(Location location, String name, String protocol,
71              int loadBalancerPort, int instancePort, Iterable<? extends NodeMetadata> nodes) {
72        checkNotNull(location, "location");
73        String region = getRegionFromLocationOrNull(location);
74  
75        Set<String> zonesDesired = ImmutableSet.copyOf(transform(nodes, new Function<NodeMetadata, String>() {
76  
77           @Override
78           public String apply(NodeMetadata from) {
79              return from.getLocation().getId();
80           }
81        }));
82  
83        logger.debug(">> creating loadBalancer(%s) in zones(%s)", name, zonesDesired);
84        try {
85           String dnsName = api.getLoadBalancerApiForRegion(region).createLoadBalancerListeningInAvailabilityZones(
86                    name,
87                    ImmutableSet.of(Listener.builder().port(loadBalancerPort).instancePort(instancePort)
88                             .protocol(Protocol.valueOf(protocol)).build()), zonesDesired);
89           logger.debug("<< created loadBalancer(%s) dnsName(%s)", name, dnsName);
90        } catch (IllegalStateException e) {
91           logger.debug("<< converging zones(%s) in loadBalancer(%s)", zonesDesired, name);
92           Set<String> currentZones = api.getLoadBalancerApi().get(name).getAvailabilityZones();
93           Set<String> zonesToAdd = Sets.difference(zonesDesired, currentZones);
94           if (zonesToAdd.size() > 0)
95              currentZones = api.getAvailabilityZoneApi().addAvailabilityZonesToLoadBalancer(zonesToAdd, name);
96           Set<String> zonesToRemove = Sets.difference(currentZones, zonesDesired);
97           if (zonesToRemove.size() > 0)
98              api.getAvailabilityZoneApi().removeAvailabilityZonesFromLoadBalancer(zonesToRemove, name);
99        }
100 
101       Set<String> instanceIds = ImmutableSet.copyOf(transform(nodes, new Function<NodeMetadata, String>() {
102 
103          @Override
104          public String apply(NodeMetadata from) {
105             return from.getProviderId();
106          }
107       }));
108 
109       logger.debug(">> converging loadBalancer(%s) to instances(%s)", name, instanceIds);
110       Set<String> registeredInstanceIds = api.getInstanceApiForRegion(region).registerInstancesWithLoadBalancer(
111                instanceIds, name);
112 
113       Set<String> instancesToRemove = filter(registeredInstanceIds, not(in(instanceIds)));
114       if (instancesToRemove.size() > 0) {
115          logger.debug(">> deregistering instances(%s) from loadBalancer(%s)", instancesToRemove, name);
116          api.getInstanceApiForRegion(region).deregisterInstancesFromLoadBalancer(instancesToRemove, name);
117       }
118       logger.debug("<< converged loadBalancer(%s) ", name);
119 
120       return converter.apply(new LoadBalancerInRegion(api.getLoadBalancerApiForRegion(region).get(name), region));
121    }
