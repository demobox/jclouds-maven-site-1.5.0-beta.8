40     public void testParseBase64InForm() {
41        Multimap<String, String> expects = LinkedListMultimap.create();
42        expects.put("Version", "2010-06-15");
43        expects.put("Action", "ModifyInstanceAttribute");
44        expects.put("Attribute", "userData");
45        expects.put("Value", "dGVzdA==");

49           expects);

52     @Test
53     public void testParseQueryToMapSingleParam() {
54        Multimap<String, String> parsedMap = parseQueryToMap("v=1.3");
55        assert parsedMap.keySet().size() == 1 : "Expected 1 key, found: " + parsedMap.keySet().size();
56        assert parsedMap.keySet().contains("v") : "Expected v to be a part of the keys";
57        String valueForV = Iterables.getOnlyElement(parsedMap.get("v"));
58        assert valueForV.equals("1.3") : "Expected the value for 'v' to be '1.3', found: " + valueForV;
59     }
