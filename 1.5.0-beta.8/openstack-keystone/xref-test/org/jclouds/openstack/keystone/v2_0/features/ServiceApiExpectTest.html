45     public void testListTenants() {
46        ServiceApi api = requestsSendResponses(
47                 keystoneAuthWithUsernameAndPassword,
48                 responseWithKeystoneAccess,
49                 authenticatedGET().endpoint(endpoint + "/v2.0/tenants").build(),
50                 HttpResponse.builder().statusCode(200)
51                          .payload(payloadFromResourceWithContentType("/tenant_list.json", APPLICATION_JSON)).build())
52                 .getServiceApi();
53        Set<Tenant> tenants = api.listTenants();
54        assertNotNull(tenants);
55        assertFalse(tenants.isEmpty());
56  
57        Set<Tenant> expected = ImmutableSet.of(Tenant.builder().name("demo").id("05d1dc7af71646deba64cfc17b81bec0")
58                 .build(), Tenant.builder().name("admin").id("7aa2e17ec29f44d193c48feaba0852cc").build());
59  
60        assertEquals(tenants, expected);
61     }
62  
63     public void testListTenantsFailNotFound() {

65                 authenticatedGET().endpoint(endpoint + "/v2.0/tenants").build(),
66                 HttpResponse.builder().statusCode(404).build()).getServiceApi();
