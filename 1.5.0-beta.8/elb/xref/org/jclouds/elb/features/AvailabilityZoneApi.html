26  /**
27   * Provides access to Amazon ELB via the Query API
28   * <p/>
29   * 
30   * @see <a href="http://docs.amazonwebservices.com/ElasticLoadBalancing/latest/APIReference" />
31   * @see AvailabilityZoneAsyncApi
32   * @author Adrian Cole

34  @Timeout(duration = 30, timeUnit = TimeUnit.SECONDS)
35  public interface AvailabilityZoneApi {
36  
37     /**
38      * Adds one or more EC2 Availability Zones to the LoadBalancer.
39      * 
40      * The LoadBalancer evenly distributes requests across all its registered Availability Zones that
41      * contain instances. As a result, the api must ensure that its LoadBalancer is appropriately
42      * scaled for each registered Availability Zone.
43      * 
44      * <h4>Note</h4>
45      * 
46      * The new EC2 Availability Zones to be added must be in the same EC2 Region as the Availability
47      * Zones for which the LoadBalancer was created.
48      * 
49      * @param zones
50      *           A list of new Availability Zones for the LoadBalancer. Each Availability Zone must
51      *           be in the same Region as the LoadBalancer.
52      * 
53      * @param loadBalancerName
54      *           The name associated with the LoadBalancer. The name must be unique within the api
55      *           AWS account.
56      * 
57      * @return An updated list of Availability Zones for the LoadBalancer.
58      */
59     Set<String> addAvailabilityZonesToLoadBalancer(Iterable<String> zones, String loadBalancerName);
60  
61     Set<String> addAvailabilityZoneToLoadBalancer(String zone, String loadBalancerName);
62  
63     /**
64      * Removes the specified EC2 Availability Zones from the set of configured Availability Zones for
65      * the LoadBalancer.
66      * 
67      * 
68      * There must be at least one Availability Zone registered with a LoadBalancer at all times. A
69      * api cannot remove all the Availability Zones from a LoadBalancer. Once an Availability Zone
70      * is removed, all the instances registered with the LoadBalancer that are in the removed
71      * Availability Zone go into the OutOfService state. Upon Availability Zone removal, the
72      * LoadBalancer attempts to equally balance the traffic among its remaining usable Availability
73      * Zones. Trying to remove an Availability Zone that was not associated with the LoadBalancer
74      * does nothing.
75      * 
76      * <h4>Note</h4>
77      * 
78      * In order for this call to be successful, the api must have created the LoadBalancer. The
79      * api must provide the same account credentials as those that were used to create the
80      * LoadBalancer.
81      * 
82      * @param zones
83      *           A list of Availability Zones to be removed from the LoadBalancer.
84      * 
85      *           <h4>Note</h4>
86      * 
87      *           There must be at least one Availability Zone registered with a LoadBalancer at all
88      *           times. The api cannot remove all the Availability Zones from a LoadBalancer.
89      *           Specified Availability Zones must be in the same Region.
90      * 
91      * 
92      * @param loadBalancerName
93      *           The name associated with the LoadBalancer. The name must be unique within the api
94      *           AWS account.
95      * 
96      * @return A list of updated Availability Zones for the LoadBalancer.
97      */
98     Set<String> removeAvailabilityZonesFromLoadBalancer(Iterable<String> zones, String loadBalancerName);
99  
100    Set<String> removeAvailabilityZoneFromLoadBalancer(String zone, String loadBalancerName);
