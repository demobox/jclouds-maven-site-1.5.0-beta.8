145       } else {
146          Injector i = Guice.createInjector(new JschSshClientModule(), new SLF4JLoggingModule());
147          SshClient.Factory factory = i.getInstance(SshClient.Factory.class);
148          SshClient connection;
149          if (Strings.emptyToNull(sshKeyFile) != null) {
150             connection = factory.create(HostAndPort.fromParts(sshHost, port), LoginCredentials.builder().user(sshUser)
151                   .privateKey(Strings2.toStringAndClose(new FileInputStream(sshKeyFile))).build());
152          } else {
153             connection = factory.create(HostAndPort.fromParts(sshHost, port),
154                   LoginCredentials.builder().user(sshUser).password(sshPass).build());
155          }
156          connection.connect();
157          return connection;
158       }

230    // Added for issue #1016.
231    // Note that some commands fail the first few attempt, but with default retries at 5 they do pass (for me locally).
232    // The error is "JSchException: channel is not opened".
233    // With the thread-pool size at 100, you get failures a lot more often.
234    @Test
235    public void testExecHostnameConcurrentlyWithSameSessions() throws Exception {
236       final SshClient client = setupClient();
237       ListeningExecutorService executor = MoreExecutors.listeningDecorator(Executors.newFixedThreadPool(10));
238       List<ListenableFuture<ExecResponse>> futures = new ArrayList<ListenableFuture<ExecResponse>>();
239       try {
240          for (int i = 0; i < 100; i++) {
241             futures.add(executor.submit(new Callable<ExecResponse>() {
242                @Override
243                public ExecResponse call() {
244                   ExecResponse response = client.exec("hostname");
245                   //System.out.println("completed (concurrently) "+count.incrementAndGet());
246                   return response;
247                   
248                }
249             }));
250          }
251          List<ExecResponse> responses = Futures.allAsList(futures).get(3000, TimeUnit.SECONDS);
252          for (ExecResponse response : responses) {
253             assertEquals(response.getError(), "");
254             assertEquals(response.getOutput().trim(), "localhost".equals(sshHost) ? InetAddress.getLocalHost().getHostName()
255                      : sshHost);
256          }
257       } finally {
258          executor.shutdownNow();
259          client.disconnect();
260       }
261    }

263    @Test
264    public void testExecChannelTakesStdinAndNoEchoOfCharsInOuputAndOutlivesClient() throws IOException {
265       SshClient client = setupClient();
266       ExecChannel response = client.execChannel("cat <<EOF");
267       client.disconnect();
268       assertEquals(response.getExitStatus().get(), null);
269       try {
270          PrintStream printStream = new PrintStream(response.getInput());
271          printStream.append("foo\n");
272          printStream.append("EOF\n");
273          printStream.close();
274          assertEquals(Strings2.toStringAndClose(response.getError()), "");
275          assertEquals(Strings2.toStringAndClose(response.getOutput()), "");
276       } finally {
277          Closeables.closeQuietly(response);
278       }
279       assertEquals(response.getExitStatus().get(), Integer.valueOf(0));
280    }
